async function getData() { try { // Pobieranie danych IP z API const ipResponse = await fetch("https://ipinfo.io/json?token=5cbb62c209322b"); if (!ipResponse.ok) throw new Error("Błąd pobierania IP"); const ipData = await ipResponse.json(); // Pobieranie lokalizacji dla IP const locationResponse = await fetch(`https://ipinfo.io/${ipData.ip}/json`); if (!locationResponse.ok) throw new Error("Błąd pobierania lokalizacji"); const locationData = await locationResponse.json(); // Poprawiony fingerprint (unikalne ID z hash SHA-256) const encoder = new TextEncoder(); const dataToHash = `${navigator.userAgent}-${ipData.ip}`; const hashBuffer = await crypto.subtle.digest("SHA-256", encoder.encode(dataToHash)); const fingerprint = btoa(String.fromCharCode(...new Uint8Array(hashBuffer))); const data = { ip: ipData.ip, location: `${locationData.city}, ${locationData.country}`, userAgent: navigator.userAgent, fingerprint: fingerprint }; console.log("Dane użytkownika:", data); return data; } catch (error) { console.error("Błąd:", error.message); return null; } } getData();
